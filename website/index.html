Hollywood.Movie = Milo.Model.extend({
  uriTemplate: Milo.UriTemplate('/movies/:id'),
  name: Milo.property('string')
})

Hollywood.Actor = Milo.Model.extend({
  uriTemplate: Milo.UriTemplate('/actors/:id'),
  firstName: Milo.property('string'),
  lastName: Milo.property('string'),
  movies: Milo.collection('Hollywood.Movie')
});

// retrieve Robert De Niro
var actor = Hollywood.Actor
       .find({ 'id': ‘robert-de-niro’ })
       .single();

// retrieve all movies with Robert De Niro
var movies = Hollywood.Actor
       .find({ 'id': ‘robert-de-niro’ })
       .get('movies').find().toArray();
